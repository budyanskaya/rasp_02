# === Тестирование Flask напрямую ===

# 1. Получение всех задач (Flask)
curl -X GET http://localhost:5001/api/tasks


# 2. Создание новой задачи
curl -X POST http://localhost:5001/api/tasks \
  -H "Content-Type: application/json" \
  -d '{"title": "Протестировать API", "status": "pending"}'


# 3. Получение задачи по ID
curl -X GET http://localhost:5001/api/tasks/1


# 4. Обновление статуса
curl -X PUT http://localhost:5001/api/tasks/1 \
  -H "Content-Type: application/json" \
  -d '{"status": "done"}'


# 5. Удаление задачи
curl -X DELETE http://localhost:5001/api/tasks/1


# === Тестирование через Nginx ===


# 6. Получение всех задач (через Nginx)
curl -X GET http://localhost/api/tasks


# 7. Создание через Nginx
curl -X POST http://localhost/api/tasks \
  -H "Content-Type: application/json" \
  -d '{"title": "Тест через Nginx", "status": "pending"}'


# 8. Проверка несуществующей задачи
curl -X GET http://localhost/api/tasks/999


# 9. Неправильный запрос (без title)
curl -X POST http://localhost/api/tasks \
  -H "Content-Type: application/json" \
  -d '{"status": "pending"}'
